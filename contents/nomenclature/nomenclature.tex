\section*{Nomenclature}
\begin{longtable}{llp{3.5in}}
	$a_t$               & \hspace{1cm} = & Action generated by a LangChain agent at step $t$.                       \\
	$C_t$               & \hspace{1cm} = & The state of a clinical case for the CMA at time $t$.                    \\
	$d_k$               & \hspace{1cm} = & The dimension of the key vectors in a self-attention mechanism.          \\
	$\mathcal{D}$       & \hspace{1cm} = & The dataset of anonymized student interactions used by the IA.           \\
	$f_{\text{AGENT}}$  & \hspace{1cm} = & The core decision function of a specified agent (STA, TCA, CMA, IA).     \\
	$G$                 & \hspace{1cm} = & The initial goal or objective for a LangChain agent.                     \\
	$H_{t-1}$           & \hspace{1cm} = & The history of a conversation up to time $t-1$.                          \\
	$H_t$               & \hspace{1cm} = & The history of actions and observations for an agent up to step $t$.     \\
	$k$                 & \hspace{1cm} = & The anonymity threshold in k-anonymity.                                  \\
	$K$                 & \hspace{1cm} = & The Key matrix in the self-attention mechanism.                          \\
	$m_t$               & \hspace{1cm} = & A message from a user at time $t$.                                       \\
	$\mu$               & \hspace{1cm} = & An aggregate metric computed by the IA.                                  \\
	$N$                 & \hspace{1cm} = & The set of nodes in a LangGraph state graph.                             \\
	$o_t$               & \hspace{1cm} = & An observation received by a LangChain agent at step $t$.                \\
	$\oplus$            & \hspace{1cm} = & The state merging operator in LangGraph.                                 \\
	$p(\cdot)$          & \hspace{1cm} = & The conditional probability distribution of a Large Language Model.      \\
	$P_{\text{intervention}}$ & \hspace{1cm} = & A personalized intervention plan generated by the TCA.             \\
	$Q$                 & \hspace{1cm} = & The Query matrix in the self-attention mechanism.                        \\
	$R_{\text{cumulative}}$ & \hspace{1cm} = & Cumulative risk score assessed by the STA.                           \\
	$R_{\text{immediate}}$  & \hspace{1cm} = & Immediate risk level assessed by Aika in real-time.                  \\
	$Res$               & \hspace{1cm} = & The set of available resources for the CMA.                              \\
	$S_t$               & \hspace{1cm} = & The state of the LangGraph at time step $t$.                             \\
	$\Delta S_i$        & \hspace{1cm} = & The state update produced by node $i$ in a LangGraph.                    \\
	$th_t$              & \hspace{1cm} = & A thought or reasoning step generated by a LangChain agent at step $t$.  \\
	$V$                 & \hspace{1cm} = & The Value matrix in the self-attention mechanism.                        \\
